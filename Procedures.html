Sub InsertIndexFirstColumn()

    Dim tbl As ListObject
    Dim i As Long
    Dim n As Long
    Dim arr() As Variant
    
    ' Reference the table (change names as needed)
    Set tbl = ActiveSheet.ListObjects("Table1")
    
    ' Insert new column as the first column
    tbl.ListColumns.Add Position:=1
    tbl.HeaderRowRange(1, 1).Value = "index"
    
    ' Number of data rows
    n = tbl.DataBodyRange.Rows.Count
    
    ' Create array starting from 0
    ReDim arr(1 To n, 1 To 1)
    
    For i = 1 To n
        arr(i, 1) = i - 1
    Next i
    
    ' Write to the new column in one operation (fast)
    tbl.ListColumns(1).DataBodyRange.Value = arr

End Sub

Move All Sheets Except Active (VBA)

Sub MoveAllExceptActive()

    Dim ws As Worksheet
    Dim keepSheet As Worksheet
    
    Application.ScreenUpdating = False
    
    ' Store the active sheet
    Set keepSheet = ActiveSheet
    
    ' Loop through all sheets
    For Each ws In ThisWorkbook.Worksheets
        
        ' Move every sheet except the active one
        If ws.Name <> keepSheet.Name Then
            ws.Move   ' moves to a new workbook
        End If
        
    Next ws
    
    Application.ScreenUpdating = True

End Sub
